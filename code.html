local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer
local Camera = workspace.CurrentCamera
local SETTINGS = {
Enabled = true,
TeamCheck = true,
MaxDistance = 1200,
BoxColor = Color3.fromRGB(255, 255, 255),
NameColor = Color3.fromRGB(220, 220, 220),
}
local drawings = {}
local function clearPlayer(player)
local cache = drawings[player]
if not cache then return end
for _, obj in pairs(cache) do
if obj.Remove then obj:Remove() end
end
drawings[player] = nil
end
local function createEsp(player)
if drawings[player] then return end
drawings[player] = {
Box = Drawing.new("Square"),
Name = Drawing.new("Text"),
Dist = Drawing.new("Text"),
}
local d = drawings[player]
d.Box.Filled = false
d.Box.Thickness = 1
d.Box.Color = SETTINGS.BoxColor
d.Box.Visible = false
d.Name.Size = 13
d.Name.Center = true
d.Name.Outline = true
d.Name.Color = SETTINGS.NameColor
d.Dist.Size = 12
d.Dist.Center = true
d.Dist.Outline = true
d.Dist.Color = SETTINGS.NameColor
end
local function update()
if not SETTINGS.Enabled then
for player in pairs(drawings) do
clearPlayer(player)
end
return
end
for _, player in ipairs(Players:GetPlayers()) do
if player ~= LocalPlayer then
local char = player.Character
local root = char and char:FindFirstChild("HumanoidRootPart")
local hum = char and char:FindFirstChildOfClass("Humanoid")
local sameTeam = SETTINGS.TeamCheck
and player.Team ~= nil
and player.Team == LocalPlayer.Team
if root and hum and hum.Health > 0 and not sameTeam then
createEsp(player)
local d = drawings[player]
local pos, onScreen = Camera:WorldToViewportPoint(root.Position)
local dist = (Camera.CFrame.Position - root.Position).Magnitude
if onScreen and dist <= SETTINGS.MaxDistance then
local scale = math.clamp(1000 / dist, 0.35, 4)
local w, h = 28 * scale, 42 * scale
d.Box.Size = Vector2.new(w, h)
d.Box.Position = Vector2.new(pos.X - w / 2, pos.Y - h / 2)
d.Box.Visible = true
d.Name.Text = player.DisplayName
d.Name.Position = Vector2.new(pos.X, pos.Y - h / 2 - 14)
d.Name.Visible = true
d.Dist.Text = string.format("[%dm]", math.floor(dist))
d.Dist.Position = Vector2.new(pos.X, pos.Y + h / 2 + 2)
d.Dist.Visible = true
else
d.Box.Visible = false
d.Name.Visible = false
d.Dist.Visible = false
end
else
clearPlayer(player)
end
end
end
end
Players.PlayerRemoving:Connect(clearPlayer)
RunService.RenderStepped:Connect(update)
